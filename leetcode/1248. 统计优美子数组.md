### 1248. 统计「优美子数组」

[题目地址](https://leetcode-cn.com/problems/count-number-of-nice-subarrays/)

```javascript

var numberOfSubarrays = function (nums, k) {
    if (k === 0 || k > nums.length) return 0;

    let arr = [],
        res = 0,
        n = nums.length;

    for (let i = 0; i < n; i++) {
        if (nums[i] % 2 == 1) {
            arr.push(i);
        }
    }

    arr.unshift(-1);
    arr.push(n);

    for (let i = 1; i + k < arr.length; i++) {
        res += (arr[i] - arr[i - 1]) * (arr[i + k] - arr[i + k - 1]);
    }

    returnres
};

```
