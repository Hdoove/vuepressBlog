### 5376. 非递增顺序的最小子序列

[]()
```javascript

/**
 * @param {number[]} nums
 * @return {number[]}
 */
var minSubsequence = function(nums) {
    
    if(nums.length === 0) return 0;
    
    nums.sort((a,b) => a-b);
    
    let sum = nums.reduce((a,b) => a+b, 0);
    
    let res = [];
    
    let sum1 = 0;
    
    while(sum1 <= sum) {
        let num = nums.pop();;
        sum1 += num;
        sum -= num;
        res.push(num);
    }
    
    return res;
    
    
};
```
