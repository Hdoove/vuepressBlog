### 560. 和为K的子数组

[题目地址](https://leetcode-cn.com/problems/subarray-sum-equals-k/submissions/)

```javascript
/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number}
 */
var subarraySum = function (nums, k) {
    if (nums.length <= 0) return 0;
    let map = new Map([
        [0, 1]
    ]);
    let sum = 0;
    let count = 0;

    for (let i = 0; i < nums.length; i++) {
        sum += nums[i];

        if (map.has(sum - k)) {
            count += map.get(sum - k);
        }

        if (!map.has(sum)) {
            map.set(sum, 1);
        } else {
            map.set(sum, map.get(sum) + 1);
        }
    }

    return count;
};
```
