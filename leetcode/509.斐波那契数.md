### 509.斐波那契数

```javascript
/**
 * @param {number} N
 * @return {number}
 * 题目地址：https://leetcode-cn.com/problems/fibonacci-number/
 */
var fib = function(N, map=new Map()) {
    if(N === 0) return 0;
    if(N === 1) return 1;

    map.set(N, fib(N-1, map)+fib(N-2, map));

    return (map.get(N-1) || fib(N-1, map)) + (map.get(N-2) || fib(N-2, map));
};
```
