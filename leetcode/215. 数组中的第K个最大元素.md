### 215. 数组中的第K个最大元素

[题目地址](https://leetcode-cn.com/problems/kth-largest-element-in-an-array/)

```javascript

/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number}
 */
var findKthLargest = function(nums, k) {
    if(nums.length === 0 || nums.length < k) return -1;

    nums = nums.sort((a,b) => a-b).reverse();

    return nums[k-1];
};

```
